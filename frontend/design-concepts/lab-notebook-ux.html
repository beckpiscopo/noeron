Technical

Updated Data Model

Episode-centric organization with rich notes and cross-entry links.

NotebookEntry

interface NotebookEntry {
  id: string
  episode_id: string // Required: belongs to episode
  entry_type: 'claim' | 'paper' | 'snippet' | 'image' | 'note'
  timestamp: string // Position in episode
  title: string
  content: string

  // User annotations
  user_notes: string // Rich text/markdown
  linked_entry_ids: string[] // Cross-references

  // Type-specific fields
  claim_id?: number
  paper_id?: string
  image_url?: string
  image_prompt?: string
  chat_exchange?: ChatExchange
}

Episode Notebook Stats

interface EpisodeNotebook {
  episode_id: string
  episode_title: string
  guest: string

  // Computed/aggregated
  entry_count: number
  note_count: number
  image_count: number
  connection_count: number // Links to other eps
  last_updated: Date
}

// Query: Get all notebooks with stats
SELECT episode_id, COUNT(*) as entries
FROM notebook_entries
GROUP BY episode_id
